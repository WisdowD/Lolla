const { createCanvas } = require('canvas'); const fs = require('fs'); const path = require('path'); const sizes = [72, 96, 128, 192, 512]; function generateIcon(size) { const canvas = createCanvas(size, size); const ctx = canvas.getContext('2d'); // Background ctx.fillStyle = '#0F0E0C'; ctx.roundRect(0, 0, size, size, size * 0.18); ctx.fill(); // Gold border accent ctx.strokeStyle = '#C9A96E'; ctx.lineWidth = size * 0.025; ctx.roundRect(size*0.06, size*0.06, size*0.88, size*0.88, size * 0.12); ctx.stroke(); // Letter L (Lollo) const cx = size / 2; const cy = size / 2; const lw = size * 0.07; const lh = size * 0.48; const serif = size * 0.18; ctx.fillStyle = '#C9A96E'; // Vertical bar of L ctx.fillRect(cx - lw*1.2, cy - lh/2, lw, lh); // Horizontal bar of L ctx.fillRect(cx - lw*1.2, cy + lh/2 - lw, serif*1.6, lw); return canvas.toBuffer('image/png'); } sizes.forEach(size => { try { const buf = generateIcon(size); fs.writeFileSync(path.join(__dirname, 'icons', `icon-${size}.png`), buf); console.log(`Generated icon-${size}.png`); } catch(e) { console.log(`Skipping icon-${size}: ${e.message}`); } });
